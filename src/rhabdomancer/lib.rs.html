1//!
2//! rhabdomancer - IDA Pro vulnerability research assistant
3//! Copyright (c) 2024-2025 Marco Ivaldi <raptor@0xdeadbeef.info>
4//!
5//! > "The road to exploitable bugs is paved with unexploitable bugs."
6//! >
7//! > -- Mark Dowd
8//!
9//! Rhabdomancer is a blazing fast IDA Pro headless plugin that locates calls to potentially
10//! insecure API functions in a binary file. Auditors can backtrace from these candidate points to
11//! find pathways allowing access to untrusted input.
12//!
13//! ## Features
14//! * Blazing fast, headless user experience courtesy of IDA Pro 9 and Binarly's idalib Rust bindings.
15//! * Support for C/C++ binary targets compiled for any architecture implemented by IDA Pro.
16//! * Bad API function call locations are printed to stdout and marked in the IDB.
17//! * Known bad API functions are grouped in tiers of badness to help prioritize the audit work.
18//!     * [BAD 0] High priority - Functions that are generally considered insecure
19//!     * [BAD 1] Medium priority - Interesting functions that should be checked for insecure use cases.
20//!     * [BAD 2] Low priority - Code paths involving these functions should be carefully checked.
21//! * The list of known bad API functions can be easily customized by editing `conf/rhabdomancer.toml`.
22//!
23//! ## Blog post
24//! * <https://security.humanativaspa.it/streamlining-vulnerability-research-with-ida-pro-and-rust>
25//!
26//! ## See also
27//! * <https://github.com/0xdea/ghidra-scripts/blob/main/Rhabdomancer.java>
28//! * <https://docs.hex-rays.com/release-notes/9_0#headless-processing-with-idalib>
29//! * <https://github.com/binarly-io/idalib>
30//! * <https://books.google.it/books/about/The_Art_of_Software_Security_Assessment.html>
31//!
32//! ## Installing
33//! The easiest way to get the latest release is via [crates.io](https://crates.io/crates/rhabdomancer):
34//! 1. Download, install, and configure IDA Pro (see <https://hex-rays.com/ida-pro>).
35//! 2. Download and extract the IDA SDK (see <https://docs.hex-rays.com/developer-guide>).
36//! 3. Install LLVM/Clang (see <https://rust-lang.github.io/rust-bindgen/requirements.html>).
37//! 4. On Linux/macOS, install as follows:
38//!     ```sh
39//!     export IDASDKDIR=/path/to/idasdk
40//!     export IDADIR=/path/to/ida # if not set, the build script will check common locations
41//!     cargo install rhabdomancer
42//!     ```
43//!    On Windows, instead, use the following commands:
44//!     ```powershell
45//!     $env:LIBCLANG_PATH="\path\to\clang+llvm\bin"
46//!     $env:PATH="\path\to\ida;$env:PATH"
47//!     $env:IDASDKDIR="\path\to\idasdk"
48//!     $env:IDADIR="\path\to\ida" # if not set, the build script will check common locations
49//!     cargo install rhabdomancer
50//!     ```
51//!
52//! ## Compiling
53//! Alternatively, you can build from [source](https://github.com/0xdea/rhabdomancer):
54//! 1. Download, install, and configure IDA Pro (see <https://hex-rays.com/ida-pro>).
55//! 2. Download and extract the IDA SDK (see <https://docs.hex-rays.com/developer-guide>).
56//! 3. Install LLVM/Clang (see <https://rust-lang.github.io/rust-bindgen/requirements.html>).
57//! 4. On Linux/macOS, compile as follows:
58//!     ```sh
59//!     git clone --depth 1 https://github.com/0xdea/rhabdomancer
60//!     cd rhabdomancer
61//!     export IDASDKDIR=/path/to/idasdk # or edit .cargo/config.toml
62//!     export IDADIR=/path/to/ida # if not set, the build script will check common locations
63//!     cargo build --release
64//!     ```
65//!    On Windows, instead, use the following commands:
66//!     ```powershell
67//!     git clone --depth 1 https://github.com/0xdea/rhabdomancer
68//!     cd rhabdomancer
69//!     $env:LIBCLANG_PATH="\path\to\clang+llvm\bin"
70//!     $env:PATH="\path\to\ida;$env:PATH"
71//!     $env:IDASDKDIR="\path\to\idasdk"
72//!     $env:IDADIR="\path\to\ida" # if not set, the build script will check common locations
73//!     cargo build --release
74//!     ```
75//!
76//! ## Usage
77//! 1. Make sure IDA Pro is properly configured with a valid license.
78//! 2. Customize the list of known bad API functions in `conf/rhabdomancer.toml` if needed.
79//! 3. Run as follows:
80//!     ```sh
81//!     rhabdomancer <binary_file>
82//!     ```
83//!    Any existing `.i64` IDB file will be updated; otherwise, a new IDB file will be created.
84//! 4. Open the resulting `.i64` IDB file with IDA Pro.
85//! 5. Select `View` > `Open subviews` > `Bookmarks`
86//! 6. Enjoy your results conveniently collected in an IDA Pro window.
87//!
88//! *Note: rhabdomancer also adds comments at marked call locations.*
89//!
90//! ## Compatibility

92//! * IDA Pro 9.0.241217 - Latest compatible: v0.3.5.
93//! * IDA Pro 9.1.250226 - Latest compatible: current version.

123/// Number of marked call locations
124static COUNTER: AtomicU32 = AtomicU32::new(0);
125
126/// Priority of bad API functions
127/// * High priority - These functions are generally considered insecure
128/// * Medium priority - These functions are interesting and should be checked for insecure use cases
129/// * Low priority - Code paths involving these functions should be carefully checked
130enum Priority {
131    High,
132    Medium,
133    Low,
134}
135
136/// List of known bad API function names organized by priority
137#[derive(serde::Deserialize)]
138struct KnownBadFunctions {
139    high: Vec<String>,
140    medium: Vec<String>,
141    low: Vec<String>,
142}
143
144impl KnownBadFunctions {
145    /// Populate the list of bad API function names from the configuration file
146    fn load() -> Result<Self, ConfigError> {
147        let path = Path::new(env!("CARGO_MANIFEST_DIR")).join("conf/rhabdomancer.toml");
148
149        println!("[*] Using configuration file `{}`", path.display());
150        Config::builder()
151            .add_source(File::from(path))
152            .build()?
153            .try_deserialize()
154    }
155
156    /// Check if a function is in the list of known bad API function names and return its priority
157    fn check_function(&self, func: &Function) -> Option<Priority> {
158        let re = Regex::new(&format!(r"^[._]?{}$", &func.name()?)).ok()?;
159
160        if self.high.iter().any(|bad| re.is_match(bad)) {
161            return Some(Priority::High);
162        }
163        if self.medium.iter().any(|bad| re.is_match(bad)) {
164            return Some(Priority::Medium);
165        }
166        if self.low.iter().any(|bad| re.is_match(bad)) {
167            return Some(Priority::Low);
168        }
169
170        None
171    }
172}

174/// List of bad API functions found in the target binary organized by priority
175struct BadFunctions<'a> {
176    high: BTreeMap<FunctionId, Function<'a>>,
177    medium: BTreeMap<FunctionId, Function<'a>>,
178    low: BTreeMap<FunctionId, Function<'a>>,
179}
180
181impl<'a> BadFunctions<'a> {
182    /// Find bad API functions in the target binary
183    fn find_all(idb: &'a IDB, bad: &KnownBadFunctions) -> Self {
184        let mut found = Self {
185            high: BTreeMap::new(),
186            medium: BTreeMap::new(),
187            low: BTreeMap::new(),
188        };
189
190        for (id, f) in idb.functions() {
191            match bad.check_function(&f) {
192                Some(Priority::High) => found.insert_function(id, f, &Priority::High),
193                Some(Priority::Medium) => found.insert_function(id, f, &Priority::Medium),
194                Some(Priority::Low) => found.insert_function(id, f, &Priority::Low),
195                None => (),
196            }
197        }
198
199        found
200    }
201
202    /// Insert a new bad API function in the list
203    fn insert_function(&mut self, id: FunctionId, func: Function<'a>, priority: &Priority) {
204        match priority {
205            Priority::High => {
206                self.high.insert(id, func);
207            }
208            Priority::Medium => {
209                self.medium.insert(id, func);
210            }
211            Priority::Low => {
212                self.low.insert(id, func);
213            }
214        }
215    }
216
217    /// Locate calls to bad API functions and mark them
218    fn locate_calls(&self, idb: &'a IDB) -> anyhow::Result<()> {
219        for f in self.high.values() {
220            Self::mark_calls(idb, f, &Priority::High)?;
221        }
222        for f in self.medium.values() {
223            Self::mark_calls(idb, f, &Priority::Medium)?;
224        }
225        for f in self.low.values() {
226            Self::mark_calls(idb, f, &Priority::Low)?;
227        }
228
229        Ok(())
230    }
231
232    /// Locate calls to the specified function and mark them
233    fn mark_calls(idb: &IDB, func: &Function, priority: &Priority) -> Result<(), IDAError> {
234        // Return an error if the function name is empty (shouldn't happen)
235        let Some(func_name) = func.name() else {
236            return Err(IDAError::ffi_with("empty function name"));
237        };
238
239        // Prepare description
240        let desc = match priority {
241            Priority::High => {
242                format!("[BAD 0] {}", func_name.trim_start_matches('.'))
243            }
244            Priority::Medium => {
245                format!("[BAD 1] {}", func_name.trim_start_matches('.'))
246            }
247            Priority::Low => {
248                format!("[BAD 2] {}", func_name.trim_start_matches('.'))
249            }
250        };
251        println!("\n{desc}");
252
253        // Traverse XREFs and mark call locations
254        idb.first_xref_to(func.start_address(), XRefQuery::ALL)
255            .map_or(Ok(()), |cur| Self::traverse_xrefs(idb, &cur, &desc))
256    }
257
258    /// Recursively traverse XREFs and mark call locations
259    fn traverse_xrefs(idb: &IDB, xref: &XRef, desc: &str) -> Result<(), IDAError> {
260        // Handle .plt indirection in ELF binaries
261        if is_in_plt(idb, xref.from()) {
262            idb.first_xref_to(
263                idb.function_at(xref.from())
264                    .map_or_else(|| BADADDR.into(), |func| func.start_address()),
265                XRefQuery::ALL,
266            )
267            .map(|thunk| Self::traverse_xrefs(idb, &thunk, desc));
268        } else if xref.is_code() {
269            // Print address with caller function name if available
270            let caller = idb.function_at(xref.from()).map_or_else(
271                || "<unknown>".into(),
272                |func| func.name().unwrap_or_else(|| "<no name>".into()),
273            );
274            println!("{:#X} in {}", xref.from(), caller);
275
276            // Add a bookmark if not already present to mark the call location
277            if !idb
278                .bookmarks()
279                .get_description(xref.from())
280                .unwrap_or_default()
281                .contains("[BAD ")
282            {
283                idb.bookmarks().mark(xref.from(), desc)?;
284                COUNTER.fetch_add(1, Ordering::Relaxed);
285            }
286
287            // Add a comment if not already present to mark the call location
288            if !idb
289                .get_cmt(xref.from())
290                .unwrap_or_default()
291                .contains("[BAD ")
292            {
293                idb.append_cmt(xref.from(), desc)?;
294            }
295        }
296
297        // Process next XREF
298        xref.next_to()
299            .map_or(Ok(()), |next| Self::traverse_xrefs(idb, &next, desc))
300    }
301}

303/// Locate calls to potentially insecure API functions in the binary file at `filepath`.
304///
305/// ## Errors
306///
307/// Returns how many call locations were marked or a generic error in case something goes wrong.
308pub fn run(filepath: &Path) -> anyhow::Result<BookmarkIndex> {
309    // Load known bad API function names from the configuration file
310    println!("[*] Loading known bad API function names");
311    let known_bad =
312        KnownBadFunctions::load().context("Failed to load known bad API function names")?;
313
314    // Open the target binary, run auto-analysis, and keep results
315    println!("[*] Trying to analyze binary file `{}`", filepath.display());
316    let idb = IDB::open_with(filepath, true, true)
317        .with_context(|| format!("Failed to analyze binary file `{}`", filepath.display()))?;
318    println!("[+] Successfully analyzed binary file");
319    println!();
320
321    // Print binary file information
322    println!("[-] Processor: {}", idb.processor().long_name(),);
323    println!("[-] Compiler: {:?}", idb.meta().cc_id());
324    println!("[-] File type: {:?}", idb.meta().filetype());
325    println!();
326
327    // Locate and mark bad API function calls in the target binary
328    println!("[*] Finding bad API function calls...");
329    BadFunctions::find_all(&idb, &known_bad)
330        .locate_calls(&idb)
331        .context("Failed to find bad API function calls")?;
332
333    println!();
334    println!("[+] Marked {COUNTER:?} new call locations");
335    println!("[+] Done processing binary file `{}`", filepath.display());
336    Ok(COUNTER.load(Ordering::Relaxed))
337}

339/// Check if an address is in the .plt segment
340fn is_in_plt(idb: &IDB, addr: Address) -> bool {
341    idb.segment_at(addr)
342        .is_some_and(|segm| segm.name().unwrap_or_default().contains("plt"))
