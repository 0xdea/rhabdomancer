19/// Number of marked call locations
20static COUNTER: AtomicU32 = AtomicU32::new(0);
21
22/// Priority of bad API functions
23/// * High priority - These functions are generally considered insecure
24/// * Medium priority - These functions are interesting and should be checked for insecure use cases
25/// * Low priority - Code paths involving these functions should be carefully checked
26enum Priority {
27    High,
28    Medium,
29    Low,
30}
31
32/// List of known bad API function names organized by priority
33#[derive(serde::Deserialize)]
34struct KnownBadFunctions {
35    high: Vec<String>,
36    medium: Vec<String>,
37    low: Vec<String>,
38}
39
40impl KnownBadFunctions {
41    /// Populate the list of bad API function names from the configuration file
42    fn load() -> Result<Self, ConfigError> {
43        let path = Path::new(env!("CARGO_MANIFEST_DIR")).join("conf/rhabdomancer.toml");
44
45        println!("[*] Using configuration file `{}`", path.display());
46        Config::builder()
47            .add_source(File::from(path))
48            .build()?
49            .try_deserialize()
50    }
51
52    /// Check if a function is in the list of known bad API function names and return its priority
53    fn check_function(&self, func: &Function) -> Option<Priority> {
54        let re = Regex::new(&format!(r"^[._]?{}$", &func.name()?)).ok()?;
55
56        if self.high.iter().any(|bad| re.is_match(bad)) {
57            return Some(Priority::High);
58        }
59        if self.medium.iter().any(|bad| re.is_match(bad)) {
60            return Some(Priority::Medium);
61        }
62        if self.low.iter().any(|bad| re.is_match(bad)) {
63            return Some(Priority::Low);
64        }
65
66        None
67    }
68}

70/// List of bad API functions found in the target binary organized by priority
71struct BadFunctions<'a> {
72    high: BTreeMap<FunctionId, Function<'a>>,
73    medium: BTreeMap<FunctionId, Function<'a>>,
74    low: BTreeMap<FunctionId, Function<'a>>,
75}
76
77impl<'a> BadFunctions<'a> {
78    /// Find bad API functions in the target binary
79    fn find_all(idb: &'a IDB, bad: &KnownBadFunctions) -> Self {
80        let mut found = Self {
81            high: BTreeMap::new(),
82            medium: BTreeMap::new(),
83            low: BTreeMap::new(),
84        };
85
86        for (id, f) in idb.functions() {
87            match bad.check_function(&f) {
88                Some(Priority::High) => found.insert_function(id, f, &Priority::High),
89                Some(Priority::Medium) => found.insert_function(id, f, &Priority::Medium),
90                Some(Priority::Low) => found.insert_function(id, f, &Priority::Low),
91                None => (),
92            }
93        }
94
95        found
96    }
97
98    /// Insert a new bad API function in the list
99    fn insert_function(&mut self, id: FunctionId, func: Function<'a>, priority: &Priority) {
100        match priority {
101            Priority::High => {
102                self.high.insert(id, func);
103            }
104            Priority::Medium => {
105                self.medium.insert(id, func);
106            }
107            Priority::Low => {
108                self.low.insert(id, func);
109            }
110        }
111    }
112
113    /// Locate calls to bad API functions and mark them
114    fn locate_calls(&self, idb: &'a IDB) -> anyhow::Result<()> {
115        for f in self.high.values() {
116            Self::mark_calls(idb, f, &Priority::High)?;
117        }
118        for f in self.medium.values() {
119            Self::mark_calls(idb, f, &Priority::Medium)?;
120        }
121        for f in self.low.values() {
122            Self::mark_calls(idb, f, &Priority::Low)?;
123        }
124
125        Ok(())
126    }
127
128    /// Locate calls to the specified function and mark them
129    fn mark_calls(idb: &IDB, func: &Function, priority: &Priority) -> Result<(), IDAError> {
130        // Return an error if the function name is empty (shouldn't happen)
131        let Some(func_name) = func.name() else {
132            return Err(IDAError::ffi_with("empty function name"));
133        };
134
135        // Prepare description
136        let desc = match priority {
137            Priority::High => {
138                format!("[BAD 0] {}", func_name.trim_start_matches('.'))
139            }
140            Priority::Medium => {
141                format!("[BAD 1] {}", func_name.trim_start_matches('.'))
142            }
143            Priority::Low => {
144                format!("[BAD 2] {}", func_name.trim_start_matches('.'))
145            }
146        };
147        println!("\n{desc}");
148
149        // Traverse XREFs and mark call locations
150        idb.first_xref_to(func.start_address(), XRefQuery::ALL)
151            .map_or(Ok(()), |cur| Self::traverse_xrefs(idb, &cur, &desc))
152    }
153
154    /// Recursively traverse XREFs and mark call locations
155    fn traverse_xrefs(idb: &IDB, xref: &XRef, desc: &str) -> Result<(), IDAError> {
156        // Handle .plt indirection in ELF binaries
157        if is_in_plt(idb, xref.from()) {
158            idb.first_xref_to(
159                idb.function_at(xref.from())
160                    .map_or_else(|| BADADDR.into(), |func| func.start_address()),
161                XRefQuery::ALL,
162            )
163            .map(|thunk| Self::traverse_xrefs(idb, &thunk, desc));
164        } else if xref.is_code() {
165            // Print address with caller function name if available
166            let caller = idb.function_at(xref.from()).map_or_else(
167                || "<unknown>".into(),
168                |func| func.name().unwrap_or_else(|| "<no name>".into()),
169            );
170            println!("{:#X} in {}", xref.from(), caller);
171
172            // Add a bookmark if not already present to mark the call location
173            if !idb
174                .bookmarks()
175                .get_description(xref.from())
176                .unwrap_or_default()
177                .contains("[BAD ")
178            {
179                idb.bookmarks().mark(xref.from(), desc)?;
180                COUNTER.fetch_add(1, Ordering::Relaxed);
181            }
182
183            // Add a comment if not already present to mark the call location
184            if !idb
185                .get_cmt(xref.from())
186                .unwrap_or_default()
187                .contains("[BAD ")
188            {
189                idb.append_cmt(xref.from(), desc)?;
190            }
191        }
192
193        // Process next XREF
194        xref.next_to()
195            .map_or(Ok(()), |next| Self::traverse_xrefs(idb, &next, desc))
196    }
197}

199/// Locate calls to potentially insecure API functions in the binary file at `filepath`.
200///
201/// ## Errors
202///
203/// Returns how many call locations were marked or a generic error in case something goes wrong.
204pub fn run(filepath: &Path) -> anyhow::Result<BookmarkIndex> {
205    // Load known bad API function names from the configuration file
206    println!("[*] Loading known bad API function names");
207    let known_bad =
208        KnownBadFunctions::load().context("Failed to load known bad API function names")?;
209
210    // Open the target binary, run auto-analysis, and keep results
211    println!("[*] Analyzing binary file `{}`", filepath.display());
212    let idb = IDB::open_with(filepath, true, true)
213        .with_context(|| format!("Failed to analyze binary file `{}`", filepath.display()))?;
214    println!("[+] Successfully analyzed binary file");
215    println!();
216
217    // Print binary file information
218    println!("[-] Processor: {}", idb.processor().long_name(),);
219    println!("[-] Compiler: {:?}", idb.meta().cc_id());
220    println!("[-] File type: {:?}", idb.meta().filetype());
221    println!();
222
223    // Locate and mark bad API function calls in the target binary
224    println!("[*] Finding bad API function calls...");
225    BadFunctions::find_all(&idb, &known_bad)
226        .locate_calls(&idb)
227        .context("Failed to find bad API function calls")?;
228
229    println!();
230    println!("[+] Marked {COUNTER:?} new call locations");
231    println!("[+] Done processing binary file `{}`", filepath.display());
232    Ok(COUNTER.load(Ordering::Relaxed))
233}

235/// Check if an address is in the .plt segment
236fn is_in_plt(idb: &IDB, addr: Address) -> bool {
237    idb.segment_at(addr)
238        .is_some_and(|segm| segm.name().unwrap_or_default().contains("plt"))
